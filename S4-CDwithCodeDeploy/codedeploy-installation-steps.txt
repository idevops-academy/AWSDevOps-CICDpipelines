#!/bin/bash

#update os
sudo yum update -y

#requirements
sudo yum install ruby -y
sudo yum install wget -y
 
cd /home/ec2-user
 
#download the setup
wget https://aws-codedeploy-us-east-2.s3.us-east-2.amazonaws.com/latest/install
chmod +x ./install
 
#install
sudo ./install auto
 
#check codedeploy service status
sudo service codedeploy-agent status
